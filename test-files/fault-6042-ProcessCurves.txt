begin settings
  version 1
  runtype calc
  frictiontype Darcy
  units SI
end settings

begin sections
  mined       84  37  0.03
end sections

begin tunnel Heysen
  back 0 portal 15  mined
  fwd  500 portal 0
end tunnel

begin plots

    begin graph
      title "My graph"
      xlabel Time (s)
      ylabel "Volume flow (m^3/s)"
      transient calc velocity Heysen@10
      Dud line of input
    end graph
  end page
end plots
